# Прототип аналітики та звітності

```plantuml
@startuml
start

title Каркас інтерфейсу системи моніторингу закупівель

:Головна сторінка;

if (Не авторизований) then (так)
  :Сторінка входу;
else (ні)
  :Панель управління;
  
  if (Управління користувачами?) then (так)
    :Список користувачів;
    fork
      :Додати користувача;
    fork again
      :Редагувати користувача;
    fork again
      :Деактивувати користувача;
    end fork
  endif
  
  if (Пошук постачальників?) then (так)
    :Результати пошуку;
    fork
      :Деталі постачальника;
    fork again
      :Порівняння постачальників;
    fork again
      :Створити заявку;
    end fork
  endif
  
  if (Аналітика та звіти?) then (так)
    :Вибір типу звіту;
    :Налаштування параметрів;
    :Генерація звіту;
    :Експорт звіту;
  endif
  
  if (Управління заявками?) then (так)
    :Список заявок;
    fork
      :Нова заявка;
        :Вибір постачальника;
        :Додавання обладнання;
        :Перевірка даних;
        :Відправка на затвердження;
    fork again
      :Перегляд заявки;
      fork
        :Затвердження заявки;
        if (Сума < 25,000 грн?) then (так)
          :Автоматичне затвердження;
        else (ні)
          :Ручне затвердження;
          if (Керівник погодив?) then (так)
            :Затверджено;
          else (ні)
            :Відхилено;
            :Редагування заявки;
          endif
        endif
      fork again
        :Редагування заявки;
      end fork
    end fork
  endif
  
endif

stop
@enduml
