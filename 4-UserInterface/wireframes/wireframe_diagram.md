# Прототип аналітики та звітності

```plantuml
@startuml
title Каркас інтерфейсу системи моніторингу закупівель (спрощена версія з вибором дій та обробкою помилки авторизації)

start
:Головна сторінка;

if (Користувач авторизований?) then (ні)
  :Сторінка входу;
  :Перевірка даних користувача;
  if (Авторизація успішна?) then (так)
    :Визначення ролі користувача;
  else (ні)
    :Повідомлення про помилку авторизації;
    :Повернення на сторінку входу;
    stop
  endif
endif

if (Роль = Системний Адміністратор) then (так)
  :Панель адміністратора;

  while (Вибір дії?) is (так)
    partition "Адміністратор" {
      if (Обрати "Управління користувачами"?) then (так)
        :Список користувачів;
        fork
          :Додати користувача;
        fork again
          :Редагувати користувача;
        fork again
          :Деактивувати користувача;
        end fork
        :Перегляд журналу аудиту;
        :Повернення до панелі адміністратора;
      else (Вихід)
        break
      endif
    }
  endwhile

elseif (Роль = Менеджер з закупівель) then (так)
  :Панель менеджера;

  while (Вибір дії?) is (так)
    partition "Менеджер" {
      if (Обрати "Пошук постачальників"?) then (так)
        :Результати пошуку;
        fork
          :Деталі постачальника;
        fork again
          :Порівняння постачальників;
        fork again
          :Створити заявку;
          :Вибір постачальника;
          :Додавання обладнання;
          :Перевірка даних;
          :Відправка на затвердження;
        end fork
        :Повернення до панелі менеджера;
      elseif (Обрати "Управління заявками"?) then (так)
        :Список заявок;
        fork
          :Нова заявка;
        fork again
          :Перегляд заявки;
          fork
            :Редагування заявки;
          fork again
            :Затвердження заявки (якщо має право);
            if (Сума < 25,000 грн?) then (так)
              :Автоматичне затвердження;
            else (ні)
              :Ручне затвердження;
              if (Керівник погодив?) then (так)
                :Затверджено;
              else (ні)
                :Відхилено;
                :Редагування заявки;
              endif
            endif
          end fork
        end fork
        :Повернення до панелі менеджера;
      else (Вихід)
        break
      endif
    }
  endwhile

elseif (Роль = Аналітик) then (так)
  :Панель аналітика;

  while (Вибір дії?) is (так)
    partition "Аналітик" {
      if (Обрати "Аналітика та звіти"?) then (так)
        :Вибір типу звіту;
        :Налаштування параметрів;
        :Генерація звіту;
        :Експорт звіту;
        :Повернення до панелі аналітика;
      else (Вихід)
        break
      endif
    }
  endwhile
endif

stop
@enduml
