## Основна діаграма доменних класів.md

```plantuml
@startuml
skinparam classAttributeIconSize 0

package "Основна предметна область" {
  class Користувач {
    - id: Long
    - ім'яКористувача: String
    - email: String
    - пароль: String
    - роль: РольКористувача
    - активний: Boolean
    - датаСтворення: LocalDateTime
    + аутентифікувати(): Boolean
    + маєДозвіл(дозвіл: String): Boolean
    + деактивувати(): void
  }

  class РольКористувача {
    <<enumeration>>
    СИСТЕМНИЙ_АДМІНІСТРАТОР
    МЕНЕДЖЕР_ЗАКУПІВЕЛЬ
    АНАЛІТИК
  }

  class Постачальник {
    - id: Long
    - назва: String
    - контактнийEmail: String
    - телефон: String
    - адреса: String
    - рейтинг: Double
    - спеціалізація: String
    - датаРеєстрації: LocalDateTime
    + розрахуватиРейтинг(): Double
    + оновитиКонтактнуІнформацію(): void
  }

  class ЗаявкаНаЗакупівлю {
    - id: Long
    - номерЗаявки: String
    - опис: String
    - загальнаСума: BigDecimal
    - статус: СтатусЗаявки
    - датаСтворення: LocalDateTime
    - створенаКим: Користувач
    + розрахуватиЗагальнуСуму(): BigDecimal
    + затвердити(): void
    + відхилити(): void
    + автоматичнеЗатвердження(): Boolean
  }

  class Обладнання {
    - id: Long
    - назва: String
    - модель: String
    - специфікації: String
    - цінаЗаОдиницю: BigDecimal
    + перевіритиСпецифікації(): Boolean
  }

  class ІсторіяВзаємодії {
    - id: Long
    - постачальник: Постачальник
    - типВзаємодії: ТипВзаємодії
    - дата: LocalDateTime
    - деталі: String
    - користувач: Користувач
    + записатиВзаємодію(): void
    + отриматиІсторію(): List<ІсторіяВзаємодії>
  }

  class ЖурналАудиту {
    - id: Long
    - дія: String
    - користувач: Користувач
    - часоваМітка: LocalDateTime
    - деталі: String
    - ipАдреса: String
    + записатиДію(): void
    + отриматиЖурнал(): List<ЖурналАудиту>
  }
}

package "Безпека" {
  class СервісАутентифікації {
    - сервісШифрування: СервісШифрування
    + увійти(ім'яКористувача: String, пароль: String): РезультатАутентифікації
    + вийти(): void
    + перевіритиТокен(токен: String): Boolean
  }

  class СервісШифрування {
    + зашифруватиДані(дані: String): String
    + розшифруватиДані(зашифрованіДані: String): String
    + хешуватиПароль(пароль: String): String
  }

  class Дозвіл {
    - назва: String
    - опис: String
    + перевірити(): Boolean
  }
}

package "Аналітика та Звітність" {
  class ГенераторЗвітів {
    - шаблон: ШаблонЗвіту
    + згенеруватиЗвіт(параметри: Map<String, Object>): Звіт
    + експортуватиВPDF(звіт: Звіт): byte[]
    + експортуватиВExcel(звіт: Звіт): byte[]
  }

  class СервісАналітики {
    - обробникДаних: ОбробникДаних
    + проаналізуватиЕфективністьПостачальників(): РезультатАналізу
    + згенеруватиТрендиЗакупівель(): ДаніТрендів
    + розрахуватиЕкономіюВитрат(): BigDecimal
  }
}

package "Сповіщення" {
  class СервісСповіщень {
    - сервісEmail: СервісEmail
    - сервісSMS: СервісSMS
    + надіслатиСповіщенняЗатвердження(заявка: ЗаявкаНаЗакупівлю): void
    + надіслатиОновленняСтатусу(користувач: Користувач, повідомлення: String): void
  }
}

' Зв'язки між класами

СервісСповіщень --> ЗаявкаНаЗакупівлю

@enduml
