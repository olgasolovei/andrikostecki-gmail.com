package "Принцип єдиної відповідальності (SRP)" {
  class СервісКористувачів {
    + зареєструватиКористувача()
    + аутентифікувати()
    + оновитиПрофіль()
  }
  
  class СервісПостачальників {
    + шукатиПостачальників()
    + порівнятиПостачальників()
    + оновитиРейтинг()
  }
  
  class СервісЗакупівель {
    + створитиЗаявку()
    + затвердитиЗаявку()
    + отриматиСтатус()
  }
}

package "Принцип відкритості/закритості (OCP)" {
  interface ГенераторЗвітів {
    + згенеруватиЗвіт()
  }
  
  class ГенераторPDFЗвітів {
    + згенеруватиЗвіт()
  }
  
  class ГенераторExcelЗвітів {
    + згенеруватиЗвіт()
  }
  
  class ГенераторHTMLЗвітів {
    + згенеруватиЗвіт()
  }
  
  class СервісЗвітності {
    - генератори: List<ГенераторЗвітів>
    + додатиГенератор()
    + згенеруватиВсіЗвіти()
  }
}

package "Принцип підстановки Лісков (LSP)" {
  class Користувач {
    - id: Long
    - імья: String
    + аутентифікувати()
    + маєДозвіл()
  }
  
  class СистемнийАдміністратор {
    - рівеньДоступу: РівеньДоступу
    + керуватиКористувачами()
    + переглянутиЖурналАудиту()
  }
  
  class МенеджерЗакупівель {
    - відділ: String
    + створитиЗаявку()
    + порівнятиПостачальників()
  }
  
  class Аналітик {
    - інструментиАналітики: List<String>
    + генеруватиЗвіти()
    + аналізуватиДані()
  }
}

package "Принцип розділення інтерфейсів (ISP)" {
  interface АутентифікаціяКористувача {
    + увійти()
    + вийти()
    + перевіритиТокен()
  }
  
  interface УправлінняКористувачами {
    + зареєструватиКористувача()
    + оновитиКористувача()
    + деактивуватиКористувача()
  }
  
  interface УправлінняРолями {
    + призначитиРоль()
    + оновитиДозволи()
  }
  
  class СервісКористувачівПовний {
    + увійти()
    + вийти()
    + перевіритиТокен()
    + зареєструватиКористувача()
    + оновитиКористувача()
    + деактивуватиКористувача()
    + призначитиРоль()
    + оновитиДозволи()
  }
}

package "Принцип інверсії залежностей (DIP)" {
  interface РепозиторійКористувачів {
    + знайтиЗаId()
    + зберегти()
    + знайтиЗаEmail()
    + знайтиВсіхАктивних()
  }
  
  interface СервісШифрування {
    + зашифруватиДані()
    + розшифруватиДані()
    + хешуватиПароль()
  }
  
  class СервісКористувачівЗалежності {
    - репозиторійКористувачів: РепозиторійКористувачів
    - сервісШифрування: СервісШифрування
    + зареєструватиКористувача()
    + аутентифікувати()
  }
  
  class JpaРепозиторійКористувачів {
    + знайтиЗаId()
    + зберегти()
    + знайтиЗаEmail()
    + знайтиВсіхАктивних()
  }
  
  class AesСервісШифрування {
    + зашифруватиДані()
    + розшифруватиДані()
    + хешуватиПароль()
  }
}
